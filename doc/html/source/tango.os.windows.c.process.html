{
    import tango.stdc.stdint;
    import tango.stdc.stddef;
}

{
    P_WAIT,
    P_NOWAIT,
    P_OVERLAY,
    P_NOWAITO,
    P_DETACH,
}

{
    WAIT_CHILD,
    WAIT_GRANDCHILD,
}

{
    extern (C) alias void function(void*) bt_fptr;
    extern (Windows) alias uint function(void*) btex_fptr;
}
