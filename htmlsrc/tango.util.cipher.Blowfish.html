            0x243f6a88u, 0x85a308d3u, 0x13198a2eu, 0x03707344u, 0xa4093822u, 0x299f31d0u,
            0x082efa98u, 0xec4e6c89u, 0x452821e6u, 0x38d01377u, 0xbe5466cfu, 0x34e90c6cu,
            0xc0ac29b7u, 0xc97c50ddu, 0x3f84d5b5u, 0xb5470917u, 0x9216d5d9u, 0x8979fb1bu
        ];
        
        enum uint BLOCK_SIZE = 8,
                  ROUNDS = 16,
                  SBOX_SIZE = 256,
                  PBOX_SIZE = 18,
                  MAX_KEY_SIZE = 56, // 448 bits
                  MIN_KEY_SIZE = 4; // 32 bits
        uint[18] P;
        uint[256] S0, S1, S2, S3;
        const(ubyte)[] workingKey;
    } // end private

    this() {}
    this(bool encrypt, ubyte[] key) {
        init(encrypt, key);
    }

    @property final override const(char[]) name()
    {
        return "Blowfish";
    }
    
    @property final override const uint blockSize()
    {
        return BLOCK_SIZE;
    }
    
    final void init(bool encrypt, ubyte[] key)
    {
        _encrypt = encrypt;
        
        size_t len = key.length;
        if (len < MIN_KEY_SIZE || len > MAX_KEY_SIZE)
            invalid(name()~": Invalid key length (requires 4-56 bytes)");
        
        workingKey = key;
        
        // Yes, this is ghetto. I know.
        _initialized = true;
        
        setup(workingKey);
        
        if (!_encrypt)
            P.reverse; // Oh yes I did.
    }
    
    private uint F(uint x)
    {
        return (((S0[(x >> 24)] 
                + S1[cast(ubyte)(x >> 16)])
                ^ S2[cast(ubyte)(x >> 8)]) 
                + S3[cast(ubyte)x]);
    }
    
    final override uint update(const(void[]) input_, void[] output_)
    {
        if (!_initialized)
            invalid(name()~": Cipher not initialized.");
        
        ubyte[] input = cast(ubyte[]) input_,
                output = cast(ubyte[]) output_;
                    
        if (input.length < BLOCK_SIZE)
            invalid (name()~": Input buffer too short");
            
        if (output.length < BLOCK_SIZE)
            invalid (name()~": Output buffer too short");
        
        uint xl = ByteConverter.BigEndian.to!(uint)(input[0..4]),
             xr = ByteConverter.BigEndian.to!(uint)(input[4..8]),
             i = 0;
        
        xl ^= P[i++];
        for (; i < ROUNDS;)
        {
            xr ^= F(xl) ^ P[i++];
            xl ^= F(xr) ^ P[i++];
        }
        xr ^= P[i];
  
        ByteConverter.BigEndian.from!(uint)(xr, output[0..4]);
        ByteConverter.BigEndian.from!(uint)(xl, output[4..8]);
        
        return BLOCK_SIZE;
    }
    
    final override void reset()
    {
        setup(workingKey);
    }
    
    private void setup(const(ubyte)[] key)
    {
        S0[] = S_INIT[0..256];
        S1[] = S_INIT[256..512];
        S2[] = S_INIT[512..768];
        S3[] = S_INIT[768..1024];
        P[] = P_INIT[];
        
        uint index = 0;

        for (int i = 0; i < PBOX_SIZE; i++)
        {
            uint x = 0;
            for (int j = 0; j < 4; j++)
            {
                x = (x << 8) | key[index++];
                if (index == key.length)
                    index = 0;
            }
            
            P[i] ^= x;
        }
        
        ubyte[] t = new ubyte[BLOCK_SIZE]; // Initialized to 0's
        for (int i = 0; i < PBOX_SIZE;)
        {
            update(t, t);
            P[i++] = ByteConverter.BigEndian.to!(uint)(t[0..4]);
            P[i++] = ByteConverter.BigEndian.to!(uint)(t[4..8]);
        }
        
        for (int i = 0; i < SBOX_SIZE;)
        {
            update(t, t);
            S0[i++] = ByteConverter.BigEndian.to!(uint)(t[0..4]);
            S0[i++] = ByteConverter.BigEndian.to!(uint)(t[4..8]);
        }
        
        for (int i = 0; i < SBOX_SIZE;)
        {
            update(t, t);
            S1[i++] = ByteConverter.BigEndian.to!(uint)(t[0..4]);
            S1[i++] = ByteConverter.BigEndian.to!(uint)(t[4..8]);
        }
        
        for (int i = 0; i < SBOX_SIZE;)
        {
            update(t, t);
            S2[i++] = ByteConverter.BigEndian.to!(uint)(t[0..4]);
            S2[i++] = ByteConverter.BigEndian.to!(uint)(t[4..8]);
        }
        
        for (int i = 0; i < SBOX_SIZE;)
        {
            update(t, t);
            S3[i++] = ByteConverter.BigEndian.to!(uint)(t[0..4]);
            S3[i++] = ByteConverter.BigEndian.to!(uint)(t[4..8]);
        }
    }

/** Some Blowfish test vectors from Schneier's site. */
debug (UnitTest)
{
    unittest
    {
        enum immutable(char)[][] test_keys = [
            "0000000000000000",
            "ffffffffffffffff",
            "57686f206973204a6f686e2047616c743f", // I don't know, do you?
            "1111111111111111",
            "0123456789abcdef",
            "1111111111111111",
            "fedcba9876543210"
        ];
             
        enum immutable(char)[][] test_plaintexts = [
            "0000000000000000",
            "ffffffffffffffff",
            "fedcba9876543210",
            "1111111111111111",
            "1111111111111111",
            "0123456789abcdef",
            "0123456789abcdef"
        ];
            
        enum immutable(char)[][] test_ciphertexts = [
            "4ef997456198dd78",
            "51866fd5b85ecb8a",
            "cc91732b8022f684",
            "2466dd878b963c9d",
            "61f9c3802281b096",
            "7d0cc630afda1ec7",
            "0aceab0fc6a0a28d"
        ];
            
        Blowfish t = new Blowfish();
        foreach (uint i, immutable(char)[] test_key; test_keys)
        {
            ubyte[] buffer = new ubyte[t.blockSize];
            char[] result;
            auto key = ByteConverter.hexDecode(test_key);
            
            // Encryption
            t.init(true, key);
            t.update(ByteConverter.hexDecode(test_plaintexts[i]), buffer);
            result = ByteConverter.hexEncode(buffer);
            assert(result == test_ciphertexts[i],
                    t.name~": ("~result~") != ("~test_ciphertexts[i]~")");
        
            // Decryption
            t.init(false, key);
            t.update(ByteConverter.hexDecode(test_ciphertexts[i]), buffer);
            result = ByteConverter.hexEncode(buffer);
            assert(result == test_plaintexts[i],
                    t.name~": ("~result~") != ("~test_plaintexts[i]~")");
        }
    }
}
