                      (S[cast(ubyte)(t2 >> 16)] << 16) ^
                      (S[cast(ubyte)(t3 >> 8)] << 8) ^
                       S[cast(ubyte) t0];
        s2 = w[i++] ^ (S[t2 >> 24] << 24) ^
                      (S[cast(ubyte)(t3 >> 16)] << 16) ^
                      (S[cast(ubyte)(t0 >> 8)] << 8) ^
                       S[cast(ubyte) t1];
        s3 = w[  i] ^ (S[t3 >> 24] << 24) ^
                      (S[cast(ubyte)(t0 >> 16)] << 16) ^
                      (S[cast(ubyte)(t1 >> 8)] << 8) ^
                       S[cast(ubyte) t2];
    }
    
    private void decryptBlock()
    {
        uint i = 4,
             r = ROUNDS >> 1,
             t0, t1, t2, t3;
             
        while (--r >= 0)
        {
            t0 = w[i++] ^ TD0[s0 >> 24] ^
                          TD1[cast(ubyte)(s3 >> 16)] ^
                          TD2[cast(ubyte)(s2 >> 8)] ^ 
                          TD3[cast(ubyte) s1];
            t1 = w[i++] ^ TD0[s1 >> 24] ^
                          TD1[cast(ubyte)(s0 >> 16)] ^
                          TD2[cast(ubyte)(s3 >> 8)]  ^ 
                          TD3[cast(ubyte) s2];
            t2 = w[i++] ^ TD0[s2 >> 24] ^
                          TD1[cast(ubyte)(s1 >> 16)] ^
                          TD2[cast(ubyte)(s0 >> 8)] ^
                          TD3[cast(ubyte) s3];
            t3 = w[i++] ^ TD0[s3 >> 24] ^
                          TD1[cast(ubyte)(s2 >> 16)] ^
                          TD2[cast(ubyte)(s1 >> 8)] ^
                          TD3[cast(ubyte) s0];

            if (r == 0)
                break;
            
            s0 = w[i++] ^ TD0[t0 >> 24] ^
                          TD1[cast(ubyte)(t3 >> 16)] ^
                          TD2[cast(ubyte)(t2 >> 8)] ^
                          TD3[cast(ubyte) t1];
            s1 = w[i++] ^ TD0[t1 >> 24] ^
                          TD1[cast(ubyte)(t0 >> 16)] ^
                          TD2[cast(ubyte)(t3 >> 8)] ^
                          TD3[cast(ubyte) t2];
            s2 = w[i++] ^ TD0[t2 >> 24] ^
                          TD1[cast(ubyte)(t1 >> 16)] ^
                          TD2[cast(ubyte)(t0 >> 8)] ^
                          TD3[cast(ubyte) t3];
            s3 = w[i++] ^ TD0[t3 >> 24] ^
                          TD1[cast(ubyte)(t2 >> 16)] ^
                          TD2[cast(ubyte)(t1 >> 8)] ^
                          TD3[cast(ubyte) t0];
        }
        
        s0 = w[i++] ^ (RS[t0 >> 24] << 24) ^
                      (RS[cast(ubyte)(t3 >> 16)] << 16) ^
                      (RS[cast(ubyte)(t2 >> 8)] << 8) ^
                       RS[cast(ubyte) t1];
        s1 = w[i++] ^ (RS[t1 >> 24] << 24) ^
                      (RS[cast(ubyte)(t0 >> 16)] << 16) ^
                      (RS[cast(ubyte)(t3 >> 8)] << 8) ^
                       RS[cast(ubyte) t2];
        s2 = w[i++] ^ (RS[t2 >> 24] << 24) ^
                      (RS[cast(ubyte)(t1 >> 16)] << 16) ^
                      (RS[cast(ubyte)(t0 >> 8)] << 8) ^
                       RS[cast(ubyte) t3];
        s3 = w[  i] ^ (RS[t3 >> 24] << 24) ^
                      (RS[cast(ubyte)(t2 >> 16)] << 16) ^
                      (RS[cast(ubyte)(t1 >> 8)] << 8) ^
                       RS[cast(ubyte) t0];
    }
    
    final override uint update(const(void[]) input_, void[] output_)
    {
        if (!_initialized)
            invalid(name()~": Cipher not initialized.");
        
        const(ubyte[]) input = cast(const(ubyte[])) input_;
        ubyte[] output = cast(ubyte[]) output_;
                    
        if (input.length < BLOCK_SIZE)
            invalid(name()~": Input buffer too short");
            
        if (output.length < BLOCK_SIZE)
            invalid(name()~": Output buffer too short");
        
        s0 = w[0] ^ ByteConverter.BigEndian.to!(uint)(input[0..4]);
        s1 = w[1] ^ ByteConverter.BigEndian.to!(uint)(input[4..8]);
        s2 = w[2] ^ ByteConverter.BigEndian.to!(uint)(input[8..12]);
        s3 = w[3] ^ ByteConverter.BigEndian.to!(uint)(input[12..16]);
             
        (_encrypt) ? encryptBlock() : decryptBlock();

        ByteConverter.BigEndian.from!(uint)(s0, output[0..4]);
        ByteConverter.BigEndian.from!(uint)(s1, output[4..8]);
        ByteConverter.BigEndian.from!(uint)(s2, output[8..12]);
        ByteConverter.BigEndian.from!(uint)(s3, output[12..16]);
        
        return BLOCK_SIZE;
    }
    
    final override void reset() {}
    
    private uint subWord(uint x)
    {
        return ((S[x>>24] << 24)             |
                (S[cast(ubyte)(x>>8)] << 8)  |
                (S[cast(ubyte)(x>>16)] << 16)|
                (S[cast(ubyte)x]));
    }

    private void setup(const(ubyte[]) key)
    {
        size_t nk = key.length / 4;
        ROUNDS = cast(uint)nk + 6;
        w = new uint[4*(ROUNDS+1)];
        
        for (size_t i = 0, j = 0; i < nk; i++, j+=4)
            w[i] = ByteConverter.BigEndian.to!(uint)(key[j..j+int.sizeof]);
        
        for (size_t i = nk; i < w.length; i++)
        {
            uint t = w[i-1];

            if (i % nk == 0)
                t = subWord(Bitwise.rotateLeft(t, 8u)) ^ RCON[(i/nk)-1];
            else if (nk > 6 && (i % nk == 4))
                t = subWord(t);
                
            w[i] = w[i-nk] ^ t;
        }
        
        if (!_encrypt)
        {
            for (uint i = 0; i <= 4*ROUNDS; i+=4)
                w[i..i+4].reverse;
            w.reverse;
            
            for (uint i = 4; i < w.length-4; i++)
            {
                w[i] = (TD0[S[w[i]>>24]]              ^
                        TD1[S[cast(ubyte)(w[i]>>16)]] ^
                        TD2[S[cast(ubyte)(w[i]>>8)]]  ^
                        TD3[S[cast(ubyte)w[i]]]);
            }
        }
    }
