module tango.util.MinMax;

/*******************************************************************************

        Return the minimum of two arguments (of the same type)

*******************************************************************************/

T min(T)(T a, T b)
{
        return a < b ? a : b;
}

/*******************************************************************************

        Return the maximum of two arguments (of the same type)

*******************************************************************************/

T max(T)(T a, T b)
{
        return a > b ? a : b;
}
